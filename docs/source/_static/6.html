N_MODELS = 2

for i in range(N_MODELS):
    print("%s/%s" % (i+1,N_MODELS))
    people = [ Person(extraversion=float(i)/150+0.5, typ=i%2) for i in range(500) ]

    [p.addact([
        make_friend(minT=2,maxT=50,alpha=0.2),
        end_friendship(maxT=2,minT=50,alpha=0.2)
    ]) for p in people];

    m = sim(people, debug=False)
    m.addmet( prob_action )
    m.addmet( percent_oversocialized )
    m.addmet( percent_lonely )
    m.addmet( unhappiness )
    
    if i==0:
        m.addmet(network)
        
    m.run(100, 0.1)
    models[i] = m
